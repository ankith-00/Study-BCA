PROGRAM - 01 : Program to read radius of a circle and to find area and circumference.

#include <stdio.h>
#define PI 3.142

void main()
{
   int r;
   double area,circum;
   clrscr(); 
     
   printf("\n Enter radius of a circle : ");  
   scanf("%d",&r);      
   area=PI * r * r;
   circum=2 * PI * r;      
   printf("\n Area of a circle = %0.2lf",area);
   printf("\n Circumference of a circle = %0.2lf",circum);      
   getch();
}

PROGRAM - 04 : program to read a number, find the sum of the digits reverse the number and check it for palindrome.

#include< stdio.h>
#include<conio.h>

void main()
{
   int num,orgnum;
   int sum=0,rev=0,rem;
   clrscr();
   
   printf("\n Enter a number : ");
   scanf("%d",&num);
   orgnum=num;
   while(num>0)
   {
      rem=num%10;
      sum=sum+rem;
      rev=rev*10+rem;
      num=num/10;
   }
   printf("\n Sum of digits   = %d",sum);
   printf("\n Reversed number = %d",rev);
   if(orgnum==rev)
      printf("\n\n Number is a Palindrome");
   else
      printf("\n\n Number is not a Palindrome");
   getch();
}

OUTPUT 
Enter a number : 21012

Sum of digits = 6
Reversed number = 21012

Number is a palindrome

PROGRAM - 07 : Program to find roots of quadratic equation using Switch-Case.

#include< stdio.h>
#include<conio.h>
#include<math.h>

void main()
{
   int a,b,c,choice;
   double
   disc,root1,root2,real,img;
   clrscr();
   
   printf("\n Enter a,b,c : ");
   scanf("%d %d %d",&a,&b,&c);
   disc=(b*b) - (4.0*a*c);
   if(disc > 0)
     choice=1;
   else if(disc < 0)
     choice=2;
   else
     choice=3;
   switch(choice)
   {
      case 1 :
      {
         printf("Real and Distinct Roots");
         root1= ( -b + sqrt(disc) ) / (2.0 * a);
         root2= ( -b - sqrt(disc) ) / (2.0 * a);
         printf("\n Root1 = %0.2lf",root1);
         printf("\n Root2 = %0.2lf",root2);
      }
      break;
      
      case 2 :
      {
         printf("Roots are complex and imaginary");
         real = -b/(2.0 * a);
         img = sqrt(abs(disc))/(2.0 * a);
         printf("\n Root1 = %0.2lf +i %0.2lf",real,img);
         printf("\n Root2 = %0.2lf -i %0.2lf",real,img);
      }
      break;
       
      case 3 :
      {
         printf("Roots are Equal");
         root1 = -b/(2.0 * a);
         root2 = -b/(2.0 * a);
         printf("\n Root1 = %0.2lf",root1);
         printf("\n Root2 = %0.2lf",root2);
      }
      break;     
      default : printf("\n Invalid inputs");
   }
   getch();
}

OUTPUT

Enter a,b,c : 1 5 2

PROGRAM - 09 : C program to remove duplicate element in single dimensional array.

#include< stdio.h>
#include<conio.h>

void main()
{
    int a[20],n,i,j,ele;
    clrscr();
    
    printf("\nEnter array limit(n) : ");
    scanf("%d",&n);
    printf("\n\n");
    for(i=0;i<n;i++)
    {
       printf("Enter A [ %d ] : ",i);
       scanf("%d",&a[i]);
    }
    
    //remove duplicate
    for(i=0;i<n-1;i++)
    {
       ele=a[i];
       for(j=i+1;j<n;j++)
       {
          if(ele==a[j] && a[j]!=-111)
          {
             printf("\n%d Duplicate entry !",a[j]);
             a[j]=-111;
           }
        }
     }
     printf("\n\nFinal Array list :");
     for(i=0;i<n;i++)
     {
       if(a[i]!=-111)
         printf("%5d",a[i]);
     }
     getch();
}

    {
       for(j=0;j<n;j++)
       {
          printf("Enter A [ %d ] [ %d ] : ",i,j);
          scanf("%d",&a[i][j]);
        }
    }
    
    printf("\nEnter B - matrix elements\n");
    for(i=0;i<n;i++)
    {
       for(j=0;j<n;j++)
       {
          printf("Enter B [ %d ] [ %d ] :  ",i,j); 
          scanf("%d",&b[i][j]);
       }
    }
    
    printf("\nAddition of matrices\n\n");
    for(i=0;i<n;i++)
    {
       for(j=0;j<n;j++)
       {
          c[i][j] = a[i][j] + b[i][j];
          printf("%4d",c[i][j]);

    printf("\nSubtraction of matrices\n\n");

       for(j=0;j<n;j++)
       {
          d[i][j] = a[i][j] - b[i][j];

PROGRAM - 15 : To read a string and to find the number of alphabets, digits, vowels, consonants, spaces and special characters.

#include< stdio.h>
#include<conio.h>

void main()
{
   char str[100],ch;
   int acount=0, dcount=0, vcount=0,ccount=0;    
   int scount=0,spcount=0,i=0;
   clrscr();

   printf("\n Enter a string : ");
   gets(str);
   while(str[i]!='\0')
   {
      if(isalpha(str[i]))
      {
         acount++;
         ch=tolower(str[i]);
         switch(ch)
         {
            case 'a':
            case 'e':
            case 'i':
            case 'o':
            case 'u': vcount++;
            break;
            default : ccount++;
         }
      }
      else if(isdigit(str[i]))
         dcount++;
      else if(isspace(str[i]))
         scount++;
      else
         spcount++;
         i=i+1;
    }
    printf("\n No. of Alphabets = %d",acount);
    printf("\n No. of Vowels = %d",vcount);
    printf("\n No. of Consonants = %d",ccount);
    printf("\n No. of Spaces = %d",scount);
    printf("\n No. of Digits = %d",dcount);
    printf("\n No. of Special symbols = %d",spcount);
    getch();
}

OUTPUT

No of Alphabet = 12
No of Vovels   = 5
No of Constants = 7 
No of Spaces  = 2
No of Digits  = 4
No of spacial symbols = 3

PROGRAM - 16 : To Swap Two Numbers using Pointers.

#include< stdio.h>
#include<conio.h>

void main()
{
   int n1,n2,*ptr1,*ptr2,temp;
   clrscr();

   printf("\nEnter two numbers : ");
   scanf("%d %d",&n1, &n2);
   printf("\nBefore swapping N1 = %d N2 = %d\n",n1,n2);
   ptr1=&n1;
   ptr2=&n2;

  //swapping
  temp=*ptr1;
  *ptr1=*ptr2;
  *ptr2=temp;
  printf("After  swapping N1 = %d N2 =%d\n",n1,n2);
  getch();
}

OUTPUT 
Enter two numbers : 10 20 

Before swapping N1 = 10  N2 = 20
After  swapping N1 = 20  N2 = 10
